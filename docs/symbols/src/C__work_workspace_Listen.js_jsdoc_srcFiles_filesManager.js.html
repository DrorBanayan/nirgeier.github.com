  7  */
  8 var FilesManager = function() {
  9 
 10     // private section
 11 
 12     /**
 13      * The array fo files that were found in the given directory
 14      *
 15      * @type {Array} type
 16      * @field The field
 17      * @private private
 18      */
 19     var files;
 20 
 21     /**
 22      * The jQuery tree plugin to display the file as tree
 23      *
 24      * @type {jQuery.tree}
 25      * @field The field
 26      * @private private
 27      */
 28     var tree;
 29 
 30     /**
 31      * Object with the different settings/defaults that the user can change
 32      *
 33      * @type {Object}
 34      * @field The field
 35      * @private private
 36      */
 37     var settings = {
 38 
 39         /** The file input field id */
 40         ID_FILE_INPUT : 'folderInput'
 41 
 42     };
 43 
 44     /**
 45      * This method will set up the events, listeners and any other bindings that we might need later on in our code
 46      *
 47      * @private private
 48      */
 49     function setBindings() {
 50 
 51         // Add the event listener to the folder selector.
 52         // Once we select the folder we want to load adn store the files in it,
 53         document.querySelector('#' + settings.ID_FILE_INPUT).addEventListener("change", function(e) {
 54             console.log('(#' + settings.ID_FILE_INPUT + ').addEventListener');
 55             console.log(tree);
 56 
 57             // Read the files and display them as tree
 58             tree.init(e);
 59 
 60             // Process the files in the directory
 61             processDirectory();
 62         }, false);
 63     }
 64 
 65     /**
 66      * Read the files in the choosen directory
 67      */
 68     function processDirectory() {
 69         console.log('processDirectory');
 70 
 71         // fetch the lists of files
 72         // The files attribute is returning a FileList object
 73         files = $('#' + settings.ID_FILE_INPUT)[0].files;
 74 
 75         // Debug - print the files metadata
 76         FilesManager.debug.printFilesInfo(files);
 77     }
 78 
 79     /** Public section **/
 80     return {
 81 
 82         /**
 83          * Init the FilesManager. <br/>
 84          * Here we will call the binding method and will init our tree
 85          *
 86          * @public
 87          * @param {String [optional]} [selector]  The selector of the dom element to display the results inside
 88          */
 89         init : function() {
 90             // Set the needed events/listeners/binding
 91             setBindings();
 92 
 93             // init the tree
 94             tree = new FilesManager.Tree('#dir-tree');
 95 
 96         },
 97 
 98         /** "API" calls */
 99 
100         /**
101          * Get the list of files in the directory
102          *
103          * @return {Array}  files   the list of the current processed files in the chosen directory
104          */
105         getFiles : function() {
106             return files;
107         }
108     }; //return;
109 }();
110 
111 /**
112  * Init the file manager whn teh page is fully loaded
113  */
114 $(document).ready(function() {
115     FilesManager.init();
116 });
